set(APP_TARGET_NAME Vine)

find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets)
qt_standard_project_setup()

file(GLOB SRC_FILE_LIST LIST_DIRECTORIES false *.cpp)
file(GLOB HEADER_FILE_LIST LIST_DIRECTORIES false *.hpp *.h)
file(GLOB RC_FILE_LIST LIST_DIRECTORIES false *.rc)

qt_add_executable(${APP_TARGET_NAME} WIN32 ${SRC_FILE_LIST} ${HEADER_FILE_LIST} ${RC_FILE_LIST})

set_target_properties(${APP_TARGET_NAME} PROPERTIES WIN32_EXECUTABLE TRUE MACOSX_BUNDLE TRUE FOLDER vi)

target_link_libraries(${APP_TARGET_NAME} PRIVATE Qt6::Core Qt6::Gui Qt6::Widgets vi::AppfwGui vi::Di)

install(TARGETS ${APP_TARGET_NAME}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR})
